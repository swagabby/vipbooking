let partyBookings = []; // 新增：存放女神聚會的現有資料 async function openBooking(id) { currentWelfare = WELFARES.find(w => w.id === id); document.getElementById('m-title').innerText = currentWelfare.name; document.getElementById('modal').style.display = 'flex'; const dateGroup = document.getElementById('date-group'); const sidInput = document.getElementById('swagger_id'); sidInput.value = ""; if (currentWelfare.name === '大神贊助') { dateGroup.innerHTML = ``; setupGrandSponsorLogic(sidInput); } else if (currentWelfare.name === '女神聚會') { // 女神聚會特殊邏輯 dateGroup.innerHTML = ``; setupPartyLogic(sidInput); } else { dateGroup.innerHTML = ``; } } /** * 女神聚會核心邏輯：偶數月場次選單 */ function setupPartyLogic(sidInput) { const select = document.getElementById('reserve_date'); sidInput.onchange = async () => { const sid = sidInput.value.trim(); if (!sid) return; select.innerHTML = ''; select.disabled = true; try { // 向 GAS 請求女神聚會分頁的資料 const res = await fetch(`${GAS_API}?action=getPartyData&v=${Date.now()}`); partyBookings = await res.json(); renderPartySession(sid); } catch (e) { alert("同步場次失敗"); } }; function renderPartySession(sid) { select.disabled = false; select.innerHTML = ''; const now = new Date(); const currentMonth = now.getMonth() + 1; // 1-12 const currentYear = now.getFullYear(); // 計算目標場次 (下一個偶數月) // 3月(奇數) -> 4月 // 4月(偶數) -> 4月 // 5月(奇數) -> 6月 let targetMonth = (currentMonth % 2 === 0) ? currentMonth : currentMonth + 1; let targetYear = currentYear; if (targetMonth > 12) { targetMonth = 2; targetYear += 1; } const sessionName = `${targetYear}年 ${targetMonth}月場次`; // --- 檢查該 Swagger ID 是否已預約該場次 --- const isOccupied = partyBookings.some(b => String(b.swagger_id).trim() === sid && b.date === sessionName ); const opt = document.createElement('option'); opt.value = sessionName; if (isOccupied) { opt.text = `${sessionName} (此主播已被預約)`; opt.disabled = true; } else { opt.text = sessionName; } select.add(opt); } } // 大神贊助邏輯 (微調為傳入 sidInput) function setupGrandSponsorLogic(sidInput) { const select = document.getElementById('reserve_date'); sidInput.onchange = async () => { const sid = sidInput.value.trim(); if (!sid) return; select.innerHTML = ''; try { const res = await fetch(`${GAS_API}?action=getGrandSponsorData&v=${Date.now()}`); grandSponsorBookings = await res.json(); renderGrandSponsorPicker(sid); } catch (e) { alert("同步失敗"); } }; function renderGrandSponsorPicker(sid) { select.disabled = false; select.innerHTML = ''; const today = new Date(); const nextMonth5th = new Date(today.getFullYear(), today.getMonth() + 1, 5); for (let i = 1; i <= 60; i++) { const d = new Date(today); d.setDate(today.getDate() + i); if (d > nextMonth5th) break; const dateStr = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`; const isOccupiedBySid = grandSponsorBookings.some(b => { const bDate = new Date(b.date); const bStr = isNaN(bDate.getTime()) ? String(b.date).trim() : `${bDate.getFullYear()}-${String(bDate.getMonth()+1).padStart(2,'0')}-${String(bDate.getDate()).padStart(2,'0')}`; return String(b.swagger_id).trim() === sid && bStr === dateStr; }); // 其他名額判定... (同 v15 版本) const opt = document.createElement('option'); opt.value = dateStr; opt.text = dateStr + (isOccupiedBySid ? " (此ID已被預約)" : ""); opt.disabled = isOccupiedBySid; select.add(opt); } } }
